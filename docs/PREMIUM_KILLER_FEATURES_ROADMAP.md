def scan_alerts_for_coin(coin_id: str, timeframe: str = '4h'):
    """
    –°–∫–∞–Ω–∏—Ä—É–µ—Ç –≤—Å–µ –∞–∫—Ç–∏–≤–Ω—ã–µ –∞–ª–µ—Ä—Ç—ã –¥–ª—è –º–æ–Ω–µ—Ç—ã.

    """

    alerts = get_active_alerts(coin_id, timeframe)

    # 2. –ü–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—É—â–∏–µ –¥–∞–Ω–Ω—ã–µ

def evaluate_conditions(conditions: dict, data: dict) -> bool:
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç —É—Å–ª–æ–≤–∏—è –∞–ª–µ—Ä—Ç–∞ –ø—Ä–æ—Ç–∏–≤ —Ç–µ–∫—É—â–∏—Ö –¥–∞–Ω–Ω—ã—Ö"""
    rules = conditions.get('rules', [])
    logic = conditions.get('logic', 'AND')

    results = []
    for rule in rules:
        indicator = rule['indicator']
        operator = rule['operator']
        value = rule['value']

        current_value = get_indicator_value(data, indicator)
        result = compare(current_value, operator, value)
        results.append(result)

    if logic == 'AND':
        return all(results)
    else:  # OR
        return any(results)
